#include <stdlib.h>
#include <string.h>

{
    int can=10;

    char bilinenler[100];

    int bulduMu;

    printf("Isminizi giriniz:");
    scanf("%s",&isim);
    printf("Merhaba %s oyun basliyor.\n",isim);

    while(can>0)
    {   bulduMu=0;
        printf("Harf tahmini:");
        scanf("%c",tahmin);

        for(i=0;i<strlen(sorulanKelime)-1;i++)
        {
            if(tahmin==sorulanKelime[i])
            {
                printf("%c harfi kelimede mevcut.",tahmin);
                bilinenler[i]=tahmin;
                bulduMu+=1;

            }

            }
        }

        if(bilinenler==sorulanKelime)
        {
            printf("%s TEBRIKLERRRR!!",sorulanKelime);
            return 0;
        }
    }

    return 0;
}
